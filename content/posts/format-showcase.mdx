## Heading 2: Code Blocks

### TypeScript Example

```ts {1,4}
type User = {
  id: string
  name: string
  roles: string[]
}

export function greet(user: User) {
  const time = new Date().toLocaleTimeString()
  return `Hello, ${user.name}! It is ${time}.`
}

console.log(greet({ id: '1', name: 'Alex', roles: ['admin'] }))
```

### JavaScript Example with Word Highlights

```js /console|error/
function compute(a, b) {
  if (typeof a !== 'number' || typeof b !== 'number') {
    console.error('Invalid inputs')
    return NaN
  }
  return a + b
}

console.log(compute(2, 3))
```
